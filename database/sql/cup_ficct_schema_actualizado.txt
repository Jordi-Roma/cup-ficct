--
-- Name: promedio_por_materia; Type: VIEW; Schema: public; Owner: -
--

CREATE VIEW public.promedio_por_materia AS
 SELECT po.id_postulacion,
    u.ci,
    u.nombre,
    u.apellido,
    m.nombre AS materia,
    round(avg(n.nota), 2) AS promedio_materia
   FROM ((((public.nota n
     JOIN public.postulacion po ON ((n.id_postulacion = po.id_postulacion)))
     JOIN public.postulante p ON ((po.id_postulante = p.id_postulante)))
     JOIN public.usuario u ON ((p.id_usuario = u.id_usuario)))
     JOIN public.materia_cup m ON ((n.id_materia = m.id_materia)))
  GROUP BY po.id_postulacion, u.ci, u.nombre, u.apellido, m.nombre;


--
-- Name: resultado_final_postulante; Type: VIEW; Schema: public; Owner: -
--

CREATE VIEW public.resultado_final_postulante AS
 SELECT po.id_postulacion,
    u.ci,
    u.nombre,
    u.apellido,
    round(avg(n.nota), 2) AS promedio_final,
        CASE
            WHEN (avg(n.nota) >= (60)::numeric) THEN 'APROBADO'::text
            ELSE 'REPROBADO'::text
        END AS estado_final
   FROM (((public.postulacion po
     JOIN public.postulante p ON ((po.id_postulante = p.id_postulante)))
     JOIN public.usuario u ON ((p.id_usuario = u.id_usuario)))
     JOIN public.nota n ON ((po.id_postulacion = n.id_postulacion)))
  GROUP BY po.id_postulacion, u.ci, u.nombre, u.apellido;
